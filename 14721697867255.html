Archive with CLI

使用几种方式去对项目进行打包操作

使用shenzhen

先clean并且根据PP文件（这里是Ad Hoc）打包

# -w workspace位置名称
# -s schema名称
# -c config名称
ipa build -w "qingxun.xcworkspace" -s "qingxun" -c "Ad Hoc" --clean
# 上传到itunesConnect
ipa distribute:itunesconnect -a dev@tupu.im -p Tupu12345 -i 915233164 --upload

打包Demo（clean - 打包 - 上传到蒲公英）

#!/bin/sh

ipa build -w "liangce.xcworkspace" -s "liangce" -c "Ad Hoc" --clean

unzip liangce.app.dSYM.zip
leancloud upload_symbol -v \
-f "./liangce.app.dSYM" \
-i "AsXGHqesNo0paA0qQdW0anXt-gzGzoHsz" \
-k "6NBEasX2Rl0487o7Lmolxtzq"

rm -rf liangce.app.dSYM
ipa distribute:pgyer -u 6f96c00c6382e5109c539246bea00e38 -a b80c23ea536df7224b2d71eb6c1df022
rm -rf liangce.ipa
rm -rf liangce.app.dSYM.zip

使用XcodeBuild 命令打包

#!/bin/sh
project=qingxun  #Project name
configuration=Ad\ Hoc   #Build configuration
xcodebuild clean -workspace $project.xcworkspace -scheme $project
xcodebuild archive -workspace $project.xcworkspace -scheme $project -configuration $configuration -archivePath $project.xcarchive
xcodebuild -exportArchive -archivePath $project.xcarchive -exportPath $project -exportFormat ipa

使用Gym打包

#!/bin/sh
project=qingxun
gym --workspace "$project.xcworkspace" --scheme "$project" --clean --use_legacy_build_api
rm -rf $project.app.dSYM.zip
